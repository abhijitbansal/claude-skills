**Files:**

**Interfaces:**
- Consumes: WS0 `SITE_*` vars; `SITE_REMOTE` env override (floorprint convention).
- Produces: `deploy-site.sh [--dry-run]` run from app repo root: splits `${SITE_DIR}` subtree → force-pushes to `${SITE_REPO}` (or `${SITE_REMOTE}`). Guards: uncommitted changes under `${SITE_DIR}` ⇒ abort; missing `SITE_REPO` and `SITE_REMOTE` ⇒ abort with setup hint; remote URLs printed with userinfo redacted. `--dry-run` prints the split SHA + target without pushing.

```bash
#!/usr/bin/env bats

load helpers

setup() {
  TMP="$(mktemp -d)"
  DEPLOY="${BATS_TEST_DIRNAME}/../../plugins/ios-dev/skills/site-pages-deploy-kit/scripts/deploy-site.sh"
  make_fixture_app "${TMP}/app"
  mkdir -p "${TMP}/app/site"
  echo '<html><body>hi</body></html>' > "${TMP}/app/site/index.html"
  git -C "${TMP}/app" add -A
  git -C "${TMP}/app" -c user.email=t@t -c user.name=t commit -qm site
  git init -q --bare "${TMP}/remote.git"
  export SITE_REMOTE="${TMP}/remote.git"
}
teardown() { rm -rf "${TMP}"; }

@test "deploys site subtree to remote main" {
  cd "${TMP}/app"
  run bash "${DEPLOY}"
  [ "$status" -eq 0 ]
  run git -C "${TMP}/remote.git" ls-tree --name-only main
  [[ "$output" == *"index.html"* ]]
}

@test "uncommitted site changes abort" {
  cd "${TMP}/app"
  echo change >> site/index.html
  run bash "${DEPLOY}"
  [ "$status" -ne 0 ]
  [[ "$output" == *"uncommitted"* ]]
}

@test "no SITE_REMOTE and no site.repo aborts with hint" {
  cd "${TMP}/app"
  unset SITE_REMOTE
  run bash "${DEPLOY}"
  [ "$status" -ne 0 ]
  [[ "$output" == *"site.repo"* ]]
}

@test "dry-run pushes nothing" {
  cd "${TMP}/app"
  run bash "${DEPLOY}" --dry-run
  [ "$status" -eq 0 ]
  run git -C "${TMP}/remote.git" ls-tree --name-only main
  [[ "$output" != *"index.html"* ]]
}

@test "userinfo in remote URL is redacted in output" {
  cd "${TMP}/app"
  export SITE_REMOTE="https://user:tok3n@github.com/x/y.git"
  run bash "${DEPLOY}" --dry-run
  [[ "$output" != *"tok3n"* ]]
}
```

Core shape:

```bash
#!/usr/bin/env bash
set -euo pipefail
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
# shellcheck source=../../_lib/load_app_config.sh
source "${SCRIPT_DIR}/../../_lib/load_app_config.sh"

DRY=0; [[ "${1:-}" == "--dry-run" ]] && DRY=1
redact() { sed -E 's#(://)[^@/]+@#\1<redacted>@#g'; }

remote="${SITE_REMOTE:-}"
if [[ -z "${remote}" ]]; then
  [[ -n "${SITE_REPO}" ]] || { echo "no SITE_REMOTE and no site.repo in app.yml — set one" >&2; exit 1; }
  remote="git@github.com:${SITE_REPO}.git"
fi
if [[ -n "$(git status --porcelain -- "${SITE_DIR}")" ]]; then
  echo "uncommitted changes under ${SITE_DIR}/ — commit first" >&2; exit 1
fi
sha="$(git subtree split --prefix "${SITE_DIR}" HEAD)"
echo "site subtree: ${sha}"
echo "target: $(echo "${remote}" | redact)"
if [[ ${DRY} -eq 1 ]]; then echo "dry-run: not pushing"; exit 0; fi
git push --force "${remote}" "${sha}:refs/heads/main"
echo "deployed ${SITE_DIR}/ → $(echo "${remote}" | redact) main"
```

**Interfaces:**
- Produces: `verify-site.sh [site-dir]` (default `${SITE_DIR}`): per-page checks, `PASS:`/`FAIL:` lines, exit 1 on any FAIL. Checks: (1) every `*.html` has `og:image` with absolute URL + `og:image:width`/`og:image:height`; (2) declared og dimensions match the actual PNG's IHDR pixels when the image exists locally; (3) CSP meta present and contains no non-self remote origins except declared `${SITE_DOMAIN}`; (4) no `<link href="http`… external fonts (self-hosted only); (5) favicon set present (`favicon.ico` + `apple-touch-icon.png`).

- [ ] **Step 1: Failing tests:** fixture site with a valid page + 32×32 PNG (generate via python3 zlib/struct one-liner in the test — write bytes directly, no ImageMagick); cases: good page PASSes; missing og:image FAILs; wrong declared height FAILs (`og:image:height` says 64, PNG is 32); Google-Fonts link FAILs; missing favicon FAILs. PNG helper for `helpers.bash`:

```bash
make_png() {  # $1 path, $2 width, $3 height — minimal valid PNG
  python3 - "$1" "$2" "$3" <<'PY'
import struct, zlib, sys
p, w, h = sys.argv[1], int(sys.argv[2]), int(sys.argv[3])
def chunk(t, d):
    c = t + d
    return struct.pack(">I", len(d)) + c + struct.pack(">I", zlib.crc32(c))
ihdr = struct.pack(">IIBBBBB", w, h, 8, 2, 0, 0, 0)
raw = b"".join(b"\x00" + b"\x00\x00\x00" * w for _ in range(h))
png = b"\x89PNG\r\n\x1a\n" + chunk(b"IHDR", ihdr) + chunk(b"IDAT", zlib.compress(raw)) + chunk(b"IEND", b"")
open(p, "wb").write(png)
PY
}
```

- [ ] **Step 3: Implement** `verify-site.sh` — bash walks `*.html`, python3 stdlib does the parsing:

```bash
#!/usr/bin/env bash
set -uo pipefail
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
APP_CONFIG_HELPERS_ONLY=1
# shellcheck source=../../_lib/load_app_config.sh
source "${SCRIPT_DIR}/../../_lib/load_app_config.sh" 2>/dev/null || true

DIR="${1:-${SITE_DIR:-site}}"
[[ -d "${DIR}" ]] || { echo "FAIL: site dir '${DIR}' not found"; exit 1; }
fails=0

python3 - "${DIR}" <<'PY' || fails=1
import os, re, struct, sys
d = sys.argv[1]
fails = 0
def png_size(path):
    with open(path, "rb") as f:
        f.seek(16)
        return struct.unpack(">II", f.read(8))
for root, _, files in os.walk(d):
    for name in files:
        if not name.endswith(".html"): continue
        page = os.path.join(root, name)
        html = open(page, encoding="utf-8", errors="replace").read()
        def meta(prop):
            m = re.search(rf'property="{prop}"\s+content="([^"]+)"', html) or \
                re.search(rf'content="([^"]+)"\s+property="{prop}"', html)
            return m.group(1) if m else None
        og = meta("og:image")
        if not og or not og.startswith("http"):
            print(f"FAIL: {page}: og:image missing or not absolute"); fails += 1
        else:
            w, h = meta("og:image:width"), meta("og:image:height")
            if not (w and h):
                print(f"FAIL: {page}: og:image:width/height missing"); fails += 1
            else:
                local = os.path.join(d, os.path.basename(og))
                if os.path.exists(local):
                    pw, ph = png_size(local)
                    if (int(w), int(h)) != (pw, ph):
                        print(f"FAIL: {page}: og declared {w}x{h} but {os.path.basename(og)} is {pw}x{ph}"); fails += 1
        if re.search(r'<link[^>]+href="http', html):
            print(f"FAIL: {page}: external <link> resource (self-host fonts/assets)"); fails += 1
        if "http-equiv=\"Content-Security-Policy\"" not in html:
            print(f"FAIL: {page}: no CSP meta"); fails += 1
        print(f"PASS: {page}" if fails == 0 else f"checked: {page}")
for icon in ("favicon.ico", "apple-touch-icon.png"):
    if not os.path.exists(os.path.join(d, icon)):
        print(f"FAIL: {icon} missing"); fails += 1
sys.exit(1 if fails else 0)
PY

exit "${fails}"
```
